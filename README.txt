ls => lista os arquivos
git add . => adiciona todos arquivos
git checkout {branch} => vai ate branch especifica 
git merge {branch} => juntar as informações com a main (branch para main)
git branch -d {branch} => apaga branch especifica
git branch => mostra todas as branchs
cd ./"pasta"/ => abre certa pasta
cd ../ => retorna para a pasta anterior
git commit -m "Nome do commit" 
git commit --amend -m "Nome do commit corrigido"
git clean => limpa arquivos temporarios

git switch => troca de branch, verifica se não há perdas
git restore --staged => restaura um arquivo para a ultima versão commitada
